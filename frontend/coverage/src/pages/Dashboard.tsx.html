import { Activity, Banknote, Landmark, LineChart, TrendingUp, Wallet } from "lucide-react";
import { Cell, Pie, PieChart, ResponsiveContainer, Tooltip } from "recharts";
 
import { MetricCard } from "../components/shared/MetricCard";
import { StateBlock } from "../components/shared/StateBlock";
import { money, percent, signedClass } from "../components/shared/format";
import { useDashboard } from "../hooks/queries";
 
const colors = ["#0f766e", "#2563eb", "#7c3aed", "#d97706", "#be123c", "#15803d", "#52525b"];
 
export function Dashboard() {
  const { data, isLoading, error } = useDashboard();
 
  if (isLoading) return <StateBlock label="載入儀表板資料中..." />;
  if (error || !data) return <StateBlock label="儀表板資料暫時無法載入" />;
 
  return (
    <div className="space-y-5">
      <header className="flex flex-wrap items-end justify-between gap-3">
        <div>
          <h1 className="text-2xl font-semibold text-ink">投資儀表板</h1>
          <p className="mt-1 text-sm text-stone-600">即時檢視 Excel 投資組合摘要</p>
        </div>
        <div className="rounded-md border border-line bg-white px-3 py-2 text-sm text-stone-600">
          本金 {money(data.principal)}
        </div>
      </header>
 
      <div className="metric-grid grid gap-4">
        <MetricCard icon={Wallet} label="帳戶淨值" value={data.account_value} />
        <MetricCard icon={Landmark} label="股票市值" value={data.stock_market_value} />
        <MetricCard icon={Banknote} label="現金部位" value={data.cash_balance} />
        <MetricCard icon={TrendingUp} label="帳戶損益" value={data.account_pnl} hint={percent(data.account_pnl_rate)} tone="signed" />
        <MetricCard icon={LineChart} label="未實現損益" value={data.unrealized_pnl} hint={percent(data.unrealized_pnl_rate)} tone="signed" />
        <MetricCard icon={Activity} label="已實現損益" value={data.realized_pnl} tone="signed" />
      </div>
 
      <div className="grid gap-4 xl:grid-cols-[1.1fr_0.9fr]">
        <section className="rounded-lg border border-line bg-white p-4 shadow-soft">
          <div className="mb-3 flex items-center justify-between">
            <h2 className="text-base font-semibold text-ink">持股配置</h2>
            <span className="text-sm text-stone-500">股利 {money(data.dividend_income)}</span>
          </div>
          <div className="h-72">
            <ResponsiveContainer width="100%" height="100%">
              <PieChart>
                <Pie data={data.holdings_pie} dataKey="market_value" nameKey="name" innerRadius={62} outerRadius={104}>
                  {data.holdings_pie.map((entry, index) => (
                    <Cell key={entry.name} fill={colors[index % colors.length]} />
                  ))}
                </Pie>
                <Tooltip formatter={(value: number) => money(value)} />
              </PieChart>
            </ResponsiveContainer>
          </div>
          <div className="grid gap-2 sm:grid-cols-2">
            {data.holdings_pie.slice(0, 6).map((item, index) => (
              <div key={item.name} className="flex items-center justify-between rounded-md bg-panel px-3 py-2 text-sm">
                <span className="flex items-center gap-2">
                  <span className="h-2.5 w-2.5 rounded-full" style={{ background: colors[index % colors.length] }} />
                  {item.name}
                </span>
                <span>{percent(item.weight)}</span>
              </div>
            ))}
          </div>
        </section>
 
        <section className="rounded-lg border border-line bg-white p-4 shadow-soft">
          <h2 className="text-base font-semibold text-ink">最近交易</h2>
          <div className="mt-3 overflow-hidden rounded-md border border-line">
            <table className="w-full text-left text-sm">
              <tbody className="divide-y divide-line">
                {data.recent_transactions.map((tx) => (
                  <tr key={tx.id} className="bg-white">
                    <td className="px-3 py-3 text-stone-500">{tx.date}</td>
                    <td className="px-3 py-3 font-medium">{tx.action}</td>
                    <td className="px-3 py-3">{tx.name}</td>
                    <td className={`px-3 py-3 text-right ${signedClass(tx.income - tx.expense)}`}>
                      {money(tx.income || tx.expense)}
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
        </section>
      </div>
    </div>
  );
}
